两个对象数组根据键值比较，查找不同项：

```js
var arr1 = [
        {
            name:"张三",
            id:56,
            sub:0,
            value:800,
            sex:"男"
        },
        {
            name:"李四",
            id:57,
            sub:1,
            value:800,
            sex:"男"
        },
        {
            name:"王五",
            id:58,
            sub:2,
            value:800,
            sex:"男"
        },
    ];
    var arr2 = [
        {
            name:"张三",
            id:56,
            sub:0,
            value:800,
            sex:"男"
        },
        {
            name:"王五",
            id:58,
            sub:2,
            value:800,
            sex:"男"
        },
    ];
    var deleteParams = function  (array1,array2) {
        var result = []
        for (var i = 0; i < array2.length; i++) {
            var obj = array2[i]
            var num = obj.id
            var isExist = false
            for (var j = 0; j < array1.length; j++) {
                var aj = array1[j]
                var n = aj.id
                if (n === num) {
                    isExist = true
                    break
                }
            }
            if (!isExist) {
                result.push(obj)
            }
        }
        return result
    }
    //根据id值查找，arr1中删除了哪一项
    console.log(deleteParams(arr2,arr1));
```
